### script for running python script on daily API call on the GCP vm ###


## install pip3
sudo apt install python3-pip

## install packages
pip3 install numpy
pip3 install pandas
pip3 install sodapy
pip3 install sqlalchemy
pip3 install pymysql

## get update
yes | sudo apt-get update

## get .py file from bucket
gsutil cp gs://crime-api-bucket/DEPA_FinalProject_DailyCrimeAPI.py .

## take a look at the file
nano DEPA_FinalProject_DailyCrimeAPI.py

## run script ad hoc
python3 DEPA_FinalProject_DailyCrimeAPI.py

## setup/access the cron tab
crontab -e

## write to crontab to execute script daily at 8:00 am (https://medium.com/@gavinwiener/how-to-schedule-a-python-script-cron-job-dea6cbf69f4e)
0 8 * * * python3 DEPA_FinalProject_DailyCrimeAPI.py
